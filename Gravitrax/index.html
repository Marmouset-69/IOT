Mais avant de faire quel que changement que ce soit, rflchissez bien lusage que vous ferez des autres pins. Par exemple, si vous utilisez un servomoteur pilot via ondes radios avec un nRF24L01+, sachez que la librairie servo.h qui permet de les piloter ne permet pas de les brancher sur nimporte quelles broches. Et justement, ceux-ci ne peuvent tre attachs quaux bornes 9 et 10 dun arduino uno, ou arduino nano. Comme quoi, il faut toujours lister tous ses besoins, avant dattribuer telle ou telle pin dentre ou sortie tel ou tel priphrique !"

    /*
  01/11/2023
  Code inspired by :
  Fichier:      HelloWorldNRF24L01-Recepteur
  Description:  Rception d'un message "Hello World" depuis un autre arduino nano, via un NRF24L01
  Auteur:       Passion-lectronique
  Librairie utilise : https://github.com/nRF24/RF24
*/

#include <SPI.h>
#include <RF24.h>

#define pinCSN  7             // On associe la broche "CSN" du NRF24L01  la sortie digitale D7 de l'arduino
#define pinCE   8             // On associe la broche "CE" du NRF24L01  la sortie digitale D8 de l'arduino
#define tunnel  "PIPE1"       // On dfinit le "nom de tunnel" (5 caractres)  travers lequel on va recevoir les donnes de l'metteur

RF24 radio(pinCE, pinCSN);    // Instanciation du NRF24L01

    const byte adresse[6] = tunnel;       // Mise au format "byte array" du nom du tunnel
char message[32];                     // Avec cette librairie, on est "limit"  32 caractres par message

    #define LED_YELLOW_PIN 5
#define LED_GREEN_PIN 6

void setup() {
  Serial.begin(9600);
  Serial.println("Rcepteur NRF24L01");
  Serial.println("");

  pinMode(LED_GREEN_PIN, OUTPUT);
  pinMode(LED_YELLOW_PIN, OUTPUT);

  for (int i=0; i < 4; i++) {
    digitalWrite(LED_GREEN_PIN, HIGH);
    delay(100);
    digitalWrite(LED_GREEN_PIN, LOW);

    digitalWrite(LED_YELLOW_PIN, HIGH);
    delay(100);
    digitalWrite(LED_YELLOW_PIN, LOW);
    Serial.println(i);
  }
    digitalWrite(LED_GREEN_PIN, HIGH);

  // Partie NRF24
    radio.begin();                      // Initialisation du module NRF24
  radio.openReadingPipe(0, adresse);  // Ouverture du tunnel en LECTURE, avec le "nom" qu'on lui a donn
    radio.setPALevel(RF24_PA_MIN);      // Slection d'un niveau "MINIMAL" pour communiquer (pas besoin d'une forte puissance, pour nos essais)
  radio.startListening();             // Dmarrage de l'coute du NRF24 (signifiant qu'on va recevoir, et non mettre quoi que ce soit, ici)
    }

    void loop() {
  // On vrifie  chaque boucle si un message est arriv
      if (radio.available()) {
  radio.read(&message, sizeof(message));                        // Si un message vient d'arriver, on le charge dans la variable "message"
        Serial.print("<< Recied: ");
  Serial.println(message);     //  et on l'affiche sur le port srie !

        // Led blink
        digitalWrite(LED_YELLOW_PIN, HIGH);
  delay(2);
  digitalWrite(LED_YELLOW_PIN, LOW);
  }
}

    Transmitter

    /*
  01/11/2023
  Code inspired by :
  Fichier:      HelloWorldNRF24L01-Emetteur
  Description:  Emission d'un "Hello World" via un NRF24L01
  Auteur:       Passion-lectronique
  Librairie utilise : https://github.com/nRF24/RF24
*/

#include <SPI.h>
#include <RF24.h>

#define pinCSN  7             // On associe la broche "CSN" du NRF24L01  la sortie digitale D7 de l'arduino
#define pinCE   8           // On associe la broche "CE" du NRF24L01  la sortie digitale D8 de l'arduino
#define tunnel  "PIPE1"       // On dfinit un "nom de tunnel" (5 caractres), pour pouvoir communiquer d'un NRF24  l'autre

RF24 radio(pinCE, pinCSN);    // Instanciation du NRF24L01

const byte adresse[6] = tunnel;               // Mise au format "byte array" du nom du tunnel
const char message[] = "Hello World !!!";     // Message  transmettre  l'autre NRF24 (32 caractres maxi, avec cette librairie)

#define LED_GREEN_PIN 5
#define LED_YELLOW_PIN 6

void setup() {
  Serial.begin(9600);

  Serial.println("Emeteur NRF24L01");

  pinMode(LED_GREEN_PIN, OUTPUT);
  pinMode(LED_YELLOW_PIN, OUTPUT);

 for (int i=0; i < 2; i++) {
    digitalWrite(LED_GREEN_PIN, HIGH);
   delay(100);
   digitalWrite(LED_GREEN_PIN, LOW);

    digitalWrite(LED_YELLOW_PIN, HIGH);
   delay(100);
   digitalWrite(LED_YELLOW_PIN, LOW);
    Serial.println(i);
  }
  digitalWrite(LED_GREEN_PIN, HIGH);

  radio.begin();                      // Initialisation du module NRF24
  radio.openWritingPipe(adresse);     // Ouverture du tunnel en CRITURE, avec le "nom" qu'on lui a donn
  radio.setPALevel(RF24_PA_MIN);      // Slection d'un niveau "MINIMAL" pour communiquer (pas besoin d'une forte puissance, pour nos essais)
  radio.stopListening();              // Arrt de l'coute du NRF24 (signifiant qu'on va mettre, et non recevoir, ici)
  }

void loop() {
  radio.write(&message, sizeof(message));     // Envoi de notre message
  Serial.print(">> Send: ");
  Serial.println(message);
  delay(5000);

  // Led blink
  digitalWrite(LED_YELLOW_PIN, HIGH);
  delay(1);
  digitalWrite(LED_YELLOW_PIN, LOW);
}
